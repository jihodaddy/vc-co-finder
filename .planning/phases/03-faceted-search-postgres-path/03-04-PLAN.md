    `src/components/search/FacetCheckboxList.tsx` — multi-select checkbox group with show-more:
    ```tsx
    'use client';
    import { useState } from 'react';
    import { useTranslations } from 'next-intl';
    import { useQueryStates } from 'nuqs';
    import { searchParsers } from '@/lib/search/query-params';
    import { Checkbox } from '@/components/ui/checkbox';
    import { cn } from '@/lib/utils';

    type Props = {
      paramKey: 'sectors'|'stage'|'region';          // nuqs param name
      items: Array<{ value: string; label: string; count?: number }>;
      defaultLimit?: number;                         // default 8 per UI-SPEC
    };

    export function FacetCheckboxList({ paramKey, items, defaultLimit = 8 }: Props) {
      const t = useTranslations('search');
      const [query, setQuery] = useQueryStates(searchParsers, { shallow: false });
      const [expanded, setExpanded] = useState(false);
      const visible = expanded ? items : items.slice(0, defaultLimit);
      const remaining = items.length - defaultLimit;
      const selected = (query[paramKey] as string[]) ?? [];

      function toggle(value: string) {
        const next = selected.includes(value)
          ? selected.filter(v => v !== value)
          : [...selected, value];
        setQuery({ [paramKey]: next, page: 1 } as any);
      }

      return (
        <>
          {visible.map(item => (
            <label
              key={item.value}
              className="flex items-center gap-2 h-8 cursor-pointer"
            >
              <Checkbox
                checked={selected.includes(item.value)}
                onCheckedChange={() => toggle(item.value)}
              />
              <span className="text-sm text-foreground">{item.label}</span>
              {typeof item.count === 'number' && (
                <span className="text-[11px] leading-tight text-muted-foreground ml-1">
                  {t('facet.count', { count: item.count })}
                </span>
              )}
            </label>
          ))}
          {items.length > defaultLimit && (
            <button
              type="button"
              onClick={() => setExpanded(x => !x)}
              className="text-sm text-muted-foreground hover:text-foreground self-start"
            >
              {expanded ? t('facet.showLess') : t('facet.showMore')}
              {!expanded && ` (${remaining})`}
            </button>
          )}
        </>
      );
    }
    ```
    Implements UI-SPEC §D-01 (show more/less ≥8 items rule), §Accessibility (label+checkbox pair), `page: 1` reset on filter change (RESEARCH §Pitfall 6 — commit immediate URL but page 1 prevents stale-page).

    `src/components/search/FacetRangeBuckets.tsx` — Employees bucket chips + "직접 입력" popover:
    ```tsx
    'use client';
    import { useState } from 'react';
    import { useTranslations } from 'next-intl';
    import { useQueryStates } from 'nuqs';
    import { searchParsers } from '@/lib/search/query-params';
    import { EMPLOYEE_BUCKETS } from '@/lib/search/types';
    import { Popover, PopoverContent, PopoverTrigger } from '@/components/ui/popover';
    import { Button } from '@/components/ui/button';
    import { Input } from '@/components/ui/input';
    import { cn } from '@/lib/utils';

    export function FacetRangeBuckets() {
      const t = useTranslations('search');
      const [query, setQuery] = useQueryStates(searchParsers, { shallow: false });
      const current = query.employees;
      const [minTxt, setMinTxt] = useState('');
      const [maxTxt, setMaxTxt] = useState('');

      function selectBucket(b: (typeof EMPLOYEE_BUCKETS)[number]) {
        setQuery({ employees: b === current ? '' : b, page: 1 } as any);
      }

      function commitCustom() {
        const min = parseInt(minTxt, 10);
        const max = parseInt(maxTxt, 10);
        if (!Number.isFinite(min) && !Number.isFinite(max)) return;
        if (Number.isFinite(min) && Number.isFinite(max) && min > max) return;
        const v = `${Number.isFinite(min) ? min : ''}-${Number.isFinite(max) ? max : ''}`;
        setQuery({ employees: v, page: 1 } as any);
      }

      return (
        <div className="flex flex-wrap gap-2">
          {EMPLOYEE_BUCKETS.map(b => {
            const selected = current === b;
            return (
              <button
                key={b}
                type="button"
                onClick={() => selectBucket(b)}
                className={cn(
                  'h-8 px-3 rounded-full border text-sm',
                  selected
                    ? 'bg-primary text-primary-foreground border-primary'
                    : 'bg-transparent border-border text-muted-foreground hover:bg-muted'
                )}
              >
                {t(`range.bucket.employees.${b}`)}
              </button>
            );
          })}
          <Popover>
            <PopoverTrigger asChild>
              <Button variant="outline" size="sm" className="h-8 px-3 rounded-full">{t('range.customCta')}</Button>
            </PopoverTrigger>
            <PopoverContent className="w-64">
              <div className="flex flex-col gap-3">
                <Input
                  type="number"
                  inputMode="numeric"
                  placeholder={t('range.from')}
                  value={minTxt}
                  onChange={e => setMinTxt(e.target.value)}
                />
                <Input
                  type="number"
                  inputMode="numeric"
                  placeholder={t('range.to')}
                  value={maxTxt}
                  onChange={e => setMaxTxt(e.target.value)}
                />
                <Button type="button" onClick={commitCustom}>{t('drawer.applyCta')}</Button>
              </div>
            </PopoverContent>
          </Popover>
        </div>
      );
    }
    ```

    `src/components/search/FacetRangeInputs.tsx` — funding/founded from-to (blur/Enter commit):
    ```tsx
    'use client';
    import { useState, useCallback } from 'react';
    import { useTranslations } from 'next-intl';
    import { useQueryStates } from 'nuqs';
    import { searchParsers } from '@/lib/search/query-params';
    import { parseKRW } from '@/lib/format/parseKRW';
    import { Input } from '@/components/ui/input';

    type Props = { paramKey: 'funding' | 'founded' };

    function parseRange(raw: string): { min: string; max: string } {
      const [a = '', b = ''] = (raw || '').split('-');
      return { min: a, max: b };
    }

    export function FacetRangeInputs({ paramKey }: Props) {
      const t = useTranslations('search');
      const [query, setQuery] = useQueryStates(searchParsers, { shallow: false });
      const { min, max } = parseRange(query[paramKey]);
      const [minTxt, setMinTxt] = useState(min);
      const [maxTxt, setMaxTxt] = useState(max);
      const [invalid, setInvalid] = useState(false);

      const commit = useCallback(() => {
        let minVal: string = '';
        let maxVal: string = '';
        if (paramKey === 'funding') {
          const mn = minTxt ? parseKRW(minTxt) : null;
          const mx = maxTxt ? parseKRW(maxTxt) : null;
          if (mn !== null && mx !== null && mn > mx) { setInvalid(true); return; }
          setInvalid(false);
          minVal = mn?.toString() ?? '';
          maxVal = mx?.toString() ?? '';
        } else {
          const mn = minTxt ? parseInt(minTxt, 10) : NaN;
          const mx = maxTxt ? parseInt(maxTxt, 10) : NaN;
          if (Number.isFinite(mn) && Number.isFinite(mx) && mn > mx) { setInvalid(true); return; }
          setInvalid(false);
          minVal = Number.isFinite(mn) ? String(mn) : '';
          maxVal = Number.isFinite(mx) ? String(mx) : '';
        }
        if (!minVal && !maxVal) {
          setQuery({ [paramKey]: '', page: 1 } as any);
        } else {
          setQuery({ [paramKey]: `${minVal}-${maxVal}`, page: 1 } as any);
        }
      }, [minTxt, maxTxt, paramKey, setQuery]);

      return (
        <div className="flex gap-2">
          <Input
            type="text"
            inputMode={paramKey === 'funding' ? 'decimal' : 'numeric'}
            placeholder={t('range.from')}
            aria-invalid={invalid}
            value={minTxt}
            onChange={e => setMinTxt(e.target.value)}
            onBlur={commit}
    ```

    ```tsx
    'use client';
    import { useTranslations } from 'next-intl';
    import { useQueryStates } from 'nuqs';
    import { searchParsers } from '@/lib/search/query-params';
    import { stageLabel } from '@/lib/format/stage';
    import { formatKRW } from '@/lib/format/currency';
    import { Badge } from '@/components/ui/badge';
    import { X } from 'lucide-react';

    type Chip = { key: string; facet: string; label: string; remove: () => void };

    export function ActiveFilterChips() {
      const t = useTranslations('search');
      const [query, setQuery] = useQueryStates(searchParsers, { shallow: false });
      const chips: Chip[] = [];

      query.sectors.forEach(v => chips.push({
        key: `sector:${v}`, facet: t('facet.sector.label'),
        label: v,
        remove: () => setQuery({ sectors: query.sectors.filter(s => s !== v), page: 1 } as any),
      }));
      query.stage.forEach(v => chips.push({
        key: `stage:${v}`, facet: t('facet.stage.label'),
        label: stageLabel(v),
        remove: () => setQuery({ stage: query.stage.filter(s => s !== v), page: 1 } as any),
      }));
      query.region.forEach(v => chips.push({
        key: `region:${v}`, facet: t('facet.region.label'),
        label: v,
        remove: () => setQuery({ region: query.region.filter(s => s !== v), page: 1 } as any),
      }));

      if (query.employees) {
        // bucket keys map to Korean labels; free-form ranges rendered via "{min}-{max}명"
        const bucketKeys = ['1_10','11_50','51_200','201_500','501_1000','1001_plus'] as const;
        const label = (bucketKeys as readonly string[]).includes(query.employees)
          ? `${t('facet.employees.label')} ${t(`range.bucket.employees.${query.employees}` as any)}`
          : (() => {
              const [mn, mx] = query.employees.split('-');
              return `${t('facet.employees.label')} ${mn || '0'}-${mx || '∞'}${t('range.unit.employees')}`;
            })();
        chips.push({
          key: 'employees', facet: t('facet.employees.label'),
          label,
          remove: () => setQuery({ employees: '', page: 1 } as any),
        });
      }

      if (query.funding) {
        const [mn, mx] = query.funding.split('-');
        const lo = mn ? formatKRW(BigInt(mn)) : '';
        const hi = mx ? formatKRW(BigInt(mx)) : '';
        const label = lo && hi
          ? `${t('facet.funding.label')} ${lo}-${hi}`
          : lo
          ? `${t('facet.funding.label')} ${lo} 이상`
          : `${t('facet.funding.label')} ${hi} 이하`;
        chips.push({
          key: 'funding', facet: t('facet.funding.label'),
          label,
          remove: () => setQuery({ funding: '', page: 1 } as any),
        });
      }

      if (query.founded) {
        const [mn, mx] = query.founded.split('-');
        const label = mn && mx
          ? `${t('facet.founded.label')} ${mn}-${mx}${t('range.unit.year')}`
          : mn
          ? `${t('facet.founded.label')} ${mn}${t('range.unit.year')} 이후`
          : `${t('facet.founded.label')} ${mx}${t('range.unit.year')} 이전`;
        chips.push({
          key: 'founded', facet: t('facet.founded.label'),
          label,
          remove: () => setQuery({ founded: '', page: 1 } as any),
        });
      }

      if (chips.length === 0) return null;

      function clearAll() {
        setQuery({
          q: '', sectors: [], stage: [], region: [],
          employees: '', funding: '', founded: '',
          page: 1,
          // Intentionally NOT clearing: view, sort, per_page (persist user prefs per D-03)
        } as any);
      }

      return (
        <ul role="group" aria-label={t('chip.activeLabel')} className="flex flex-wrap gap-2 items-center">
          {chips.map(c => (
            <li key={c.key}>
              <Badge variant="secondary" className="inline-flex items-center gap-1">
                <span>{c.label}</span>
                <button
                  type="button"
                  aria-label={t('chip.remove', { label: c.label })}
                  onClick={c.remove}
                  className="inline-flex h-4 w-4 items-center justify-center rounded-full hover:bg-muted-foreground/20"
                >
                  <X className="h-3 w-3" />
                </button>
              </Badge>
            </li>
          ))}
          <li>
            <button
              type="button"
              onClick={clearAll}
              className="text-sm text-muted-foreground underline-offset-4 hover:text-foreground hover:underline"
            >
              {t('chip.clearAll')}
            </button>
          </li>
        </ul>
      );
    }
    ```
