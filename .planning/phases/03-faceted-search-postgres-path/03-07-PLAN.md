    ```typescript
    #!/usr/bin/env node
    /* eslint-disable no-console */
    /**
     * Synthetic 5k-row data generator for Phase 3 load testing.
     * Marker: slug LIKE 'synth-%' (cleanup via purge-synthetic.ts).
     * Distributions per RESEARCH §Pitfall 4.
     *
     * Usage:
     *   npx tsx scripts/search/generate-synthetic.ts [count=5000]
     *
     * Requires: SUPABASE_SERVICE_ROLE_KEY in .env.local (writes canonical tables).
     */
    import { createClient } from '@supabase/supabase-js';
    import 'dotenv/config';

    const TARGET = Number(process.argv[2] ?? 5000);
    const BATCH  = 500;

    const SECTORS = ['핀테크','AI','커머스','헬스케어','모빌리티','프롭테크','에듀테크','푸드테크','인슈어테크','게임','콘텐츠','SaaS','B2B','O2O','바이오','로봇','팹리스','블록체인','ESG','기타'];
    const STAGES = ['pre_a','seed','series_a','series_a','series_a','series_b','series_b','series_c','series_d','bridge','safe','grant']; // weighted
    const REGIONS_ADDR: Array<[string, number]> = [
      ['서울특별시 강남구 테헤란로 152', 65],
      ['경기도 성남시 분당구 판교역로 235', 20],
      ['부산광역시 해운대구 센텀동로 71', 5],
      ['대구광역시 수성구 동대구로 465', 3],
      ['인천광역시 연수구 송도과학로 32', 3],
      ['광주광역시 북구 첨단과기로 123', 2],
      ['대전광역시 유성구 대학로 291', 2],
    ];

    function pickWeighted<T>(pairs: Array<[T, number]>): T {
      const total = pairs.reduce((s, [, w]) => s + w, 0);
      let r = Math.random() * total;
      for (const [v, w] of pairs) { r -= w; if (r <= 0) return v; }
      return pairs[0][0];
    }
    function pickOne<T>(arr: readonly T[]): T { return arr[Math.floor(Math.random() * arr.length)]; }
    function randInt(lo: number, hi: number) { return lo + Math.floor(Math.random() * (hi - lo + 1)); }
    // power-law KRW amount: median 10억, max ~5조
    function powerLawAmount(): bigint {
      const p = Math.pow(Math.random(), 2.2);
      return BigInt(Math.floor(p * 5_000_000_000_000 + 100_000_000));
    }

    async function main() {
      const url = process.env.NEXT_PUBLIC_SUPABASE_URL;
      const key = process.env.SUPABASE_SERVICE_ROLE_KEY;
      if (!url || !key) throw new Error('NEXT_PUBLIC_SUPABASE_URL + SUPABASE_SERVICE_ROLE_KEY required in .env.local');
      const supa = createClient(url, key, { auth: { persistSession: false } });

      // Use manual_curation data source
      const { data: src } = await supa.from('data_sources').select('id').eq('slug', 'manual_curation').single();
      if (!src) throw new Error('manual_curation data_source missing — run seed first');
      const sourceId = src.id as string;

      console.log(`[synth] generating ${TARGET} synthetic companies…`);
      let inserted = 0;
      for (let batchStart = 0; batchStart < TARGET; batchStart += BATCH) {
        const rows = [];
        const end = Math.min(batchStart + BATCH, TARGET);
        for (let i = batchStart; i < end; i++) {
          const foundedYear = randInt(1990, 2025);
          rows.push({
            slug: `synth-${String(i).padStart(5, '0')}`,
            display_name_ko: `합성기업${i}`,
            display_name_en: `Synthetic Co ${i}`,
            legal_name: `주식회사 합성${i}`,
            sector: pickOne(SECTORS),
            region: 'KR',
            hq_address: pickWeighted(REGIONS_ADDR),
            founded_at: `${foundedYear}-01-01`,
            description_ko: `합성 회사 ${i} — 로드 테스트 전용`,
            website_url: null,
            logo_url: null,
            status: 'active',
            source_id: sourceId,
          });
        }
        const { error } = await supa.from('companies').upsert(rows, { onConflict: 'slug' });
        if (error) { console.error(error); process.exit(2); }
        inserted += rows.length;
        process.stdout.write(`[synth] companies ${inserted}/${TARGET}\r`);
      }
      console.log(`\n[synth] companies done`);

      // Fetch ids for aliases + funding_rounds + company_facts insertion
      const { data: allRows } = await supa.from('companies')
        .select('id, slug').ilike('slug', 'synth-%').limit(TARGET);
      const byId = allRows ?? [];

      console.log(`[synth] generating funding rounds (0-3 per company)…`);
      const fundingBatch = [];
      for (const c of byId) {
        const n = randInt(0, 3);
        for (let k = 0; k < n; k++) {
          fundingBatch.push({
            company_id: c.id,
            stage: pickOne(STAGES),
            amount_minor: powerLawAmount().toString(),
            currency_code: 'KRW',
            announced_at: `${randInt(2015, 2025)}-${String(randInt(1,12)).padStart(2,'0')}-${String(randInt(1,28)).padStart(2,'0')}`,
            source_id: sourceId,
          });
          if (fundingBatch.length >= BATCH) {
            const { error } = await supa.from('funding_rounds').insert(fundingBatch);
            if (error) throw error;
            fundingBatch.length = 0;
          }
        }
      }
      if (fundingBatch.length) {
        const { error } = await supa.from('funding_rounds').insert(fundingBatch);
        if (error) throw error;
      }
      console.log(`[synth] funding rounds done`);

      console.log(`[synth] generating aliases (1 brand alias per company)…`);
      const aliasBatch = [];
      for (const c of byId) {
        aliasBatch.push({
          company_id: c.id,
          alias: `Brand${c.slug.replace('synth-','')}`,
          alias_type: 'brand',
          source_id: sourceId,
        });
        if (aliasBatch.length >= BATCH) {
          const { error } = await supa.from('aliases').insert(aliasBatch);
          if (error) throw error;
          aliasBatch.length = 0;
        }
      }
      if (aliasBatch.length) {
        const { error } = await supa.from('aliases').insert(aliasBatch);
        if (error) throw error;
      }
      console.log(`[synth] aliases done`);

      console.log(`[synth] generating employees facts (1 per ~60% of companies)…`);
      const factsBatch = [];
      for (const c of byId) {
        if (Math.random() > 0.4) {
          factsBatch.push({
            company_id: c.id,
            fact_type: 'employees',
            fact_value: { count: Math.floor(Math.pow(Math.random(), 2) * 5000 + 1) },
            observed_at: `2025-01-01`,
            source_id: sourceId,
          });
          if (factsBatch.length >= BATCH) {
            const { error } = await supa.from('company_facts').insert(factsBatch);
            if (error) throw error;
            factsBatch.length = 0;
          }
        }
      }
      if (factsBatch.length) {
        const { error } = await supa.from('company_facts').insert(factsBatch);
        if (error) throw error;
      }
      console.log(`[synth] facts done`);
      console.log(`[synth] DONE — ${TARGET} synthetic companies active`);
    }

    main().catch(e => { console.error(e); process.exit(1); });
    ```

    ```typescript
    #!/usr/bin/env node
    /* eslint-disable no-console */
    import { createClient } from '@supabase/supabase-js';
    import 'dotenv/config';

    async function main() {
      const url = process.env.NEXT_PUBLIC_SUPABASE_URL!;
      const key = process.env.SUPABASE_SERVICE_ROLE_KEY!;
      if (!url || !key) throw new Error('missing env');
      const supa = createClient(url, key, { auth: { persistSession: false } });

      const { data: rows } = await supa.from('companies').select('id').ilike('slug', 'synth-%');
      const ids = (rows ?? []).map(r => r.id);
      console.log(`[purge] found ${ids.length} synth rows`);
      if (ids.length === 0) return;

      // Cascade-delete: children first
      for (const table of ['funding_rounds','aliases','company_facts'] as const) {
        const { error } = await supa.from(table).delete().in('company_id', ids);
        if (error) console.error(`[purge] ${table} error:`, error);
      }
      const { error } = await supa.from('companies').delete().in('id', ids);
      if (error) console.error('[purge] companies error:', error);
      console.log('[purge] done');
    }

    main().catch(e => { console.error(e); process.exit(1); });
    ```

    ```typescript
    #!/usr/bin/env node
    /* eslint-disable no-console */
    /**
     * Phase 3 Load Test — SRCH-05 / ROADMAP SC #4.
     * Runs 100 search queries against searchAdapter, records p50/p95/p99 latency.
     * Assumes 5k synthetic rows present (run `npm run synth:gen` first).
     *
     * Exit 0 if p95 < 1000ms.
     * Exit 1 if p95 >= 1000ms (Plan B documented in REPORT.md).
     *
     * Usage: npm run test:load
     */
    import { searchAdapter } from '@/lib/search/adapter';
    import { makeSearchQuery } from '../unit/search-fixtures';
    import { writeFileSync } from 'node:fs';

    const SECTORS_POOL = ['핀테크','AI','커머스','헬스케어','모빌리티'];
    const STAGES_POOL = ['seed','series_a','series_b','series_c'];
    const REGIONS_POOL = ['서울','경기','부산'];

    function pick<T>(arr: readonly T[]) { return arr[Math.floor(Math.random() * arr.length)]; }

    function buildQueryMix(): Array<ReturnType<typeof makeSearchQuery>> {
      const out: Array<ReturnType<typeof makeSearchQuery>> = [];
      // 20 single-facet
      for (let i = 0; i < 20; i++) out.push(makeSearchQuery({ sectors: [pick(SECTORS_POOL)] }));
      // 30 3-facet
      for (let i = 0; i < 30; i++) out.push(makeSearchQuery({
        sectors: [pick(SECTORS_POOL)],
        stage: [pick(STAGES_POOL)],
        region: [pick(REGIONS_POOL)],
    ```
