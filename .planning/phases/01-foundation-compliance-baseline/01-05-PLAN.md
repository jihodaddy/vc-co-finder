import 'server-only';

/**
 * Provenance metadata attached to every fact-bearing row read.
 * Per TRUST-03 + ARCHITECTURE.md Pattern "Provenance as First-Class".
 */
export type SourceMeta = {
  sourceId: string;
  sourceType: 'dart' | 'manual' | 'user_submission' | 'press_release' | 'vc_portfolio' | 'news_mention' | 'kstartup';
  sourceUrl: string | null;
  fetchedAt: string;           // ISO-8601
  lastVerifiedAt: string;      // ISO-8601 — distinct from updated_at (TRUST-02)
  confidence: number | null;
};

export type WithMeta<T> = T & { _meta: SourceMeta };

/**
 * Wraps a row with its provenance. Phase 2's lib/data/companies.ts etc. will call
 * this after JOINing data_sources.
 */
export function attachSource<T>(row: T, source: SourceMeta): WithMeta<T> {
  return { ...row, _meta: source };
}

/**
 * Convenience to attach source to an array of rows sharing one source.
 */
export function attachSourceAll<T>(rows: T[], source: SourceMeta): WithMeta<T>[] {
  return rows.map((r) => attachSource(r, source));
}
