```sql
-- Enable RLS on all canonical tables. Anon SELECT permitted; INSERT/UPDATE/DELETE
-- blocked for anon AND authenticated. Service-role bypasses RLS by default.
-- Per D-03.1 + ARCHITECTURE.md Pattern 5.

DO $$
DECLARE
  t TEXT;
BEGIN
  FOREACH t IN ARRAY ARRAY[
    'data_sources',
    'companies',
    'aliases',
    'company_identifiers',
    'company_relations',
    'funding_rounds',
    'investors',
    'round_investors',
    'persons',
    'person_roles',
    'company_facts',
    'news_mentions'
  ]
  LOOP
    EXECUTE format('ALTER TABLE public.%I ENABLE ROW LEVEL SECURITY', t);
    EXECUTE format('ALTER TABLE public.%I FORCE ROW LEVEL SECURITY', t);  -- apply to table owner too

    -- Permissive SELECT for anon + authenticated (anon browsing is core UX per D-01.4).
    -- Only non-deleted rows visible.
    EXECUTE format($p$CREATE POLICY canonical_select_public ON public.%I FOR SELECT TO anon, authenticated USING (deleted_at IS NULL)$p$, t);

    -- Explicit block on INSERT/UPDATE/DELETE for anon + authenticated (service-role bypasses).
    EXECUTE format($p$CREATE POLICY canonical_no_insert ON public.%I FOR INSERT TO anon, authenticated WITH CHECK (false)$p$, t);
    EXECUTE format($p$CREATE POLICY canonical_no_update ON public.%I FOR UPDATE TO anon, authenticated USING (false) WITH CHECK (false)$p$, t);
    EXECUTE format($p$CREATE POLICY canonical_no_delete ON public.%I FOR DELETE TO anon, authenticated USING (false)$p$, t);
  END LOOP;
END $$;

-- audit_log: ONLY service-role can read/write. Authenticated + anon blocked.
-- Forensic integrity — users (even admins) must use admin API, not direct SELECT.
-- Plan 4b admin UI reads via service-role server action.
ALTER TABLE public.audit_log ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.audit_log FORCE ROW LEVEL SECURITY;
CREATE POLICY audit_log_no_read ON public.audit_log FOR SELECT TO anon, authenticated USING (false);
CREATE POLICY audit_log_no_insert ON public.audit_log FOR INSERT TO anon, authenticated WITH CHECK (false);
CREATE POLICY audit_log_no_update ON public.audit_log FOR UPDATE TO anon, authenticated USING (false) WITH CHECK (false);
CREATE POLICY audit_log_no_delete ON public.audit_log FOR DELETE TO anon, authenticated USING (false);

COMMENT ON TABLE public.audit_log IS 'RLS: service-role only. Triggers bypass RLS (they run in definer context of fn_audit_log_write).';
```

```sql
-- TG_ARGV[0] tags the write source. Without a session-level setting, service-role ETL

  t TEXT;
BEGIN

    'data_sources',
    'companies',
    'aliases',
    'company_identifiers',
    'company_relations',
    'funding_rounds',
    'investors',
    'round_investors',
    'persons',
    'person_roles',
    'company_facts',
    'news_mentions'
  ]
  LOOP

      'CREATE TRIGGER trg_audit_%I
         AFTER INSERT OR UPDATE OR DELETE ON public.%I

      'CREATE TRIGGER trg_touch_updated_at_%I
```

```ts
import 'server-only';
import { createClient } from '@/lib/supabase/server';
import { cache } from 'react';

export type SessionUser = {
  id: string;
  email: string | null;
  role: 'user' | 'editor' | 'admin';
};

/**
 * Reads the authenticated user from the session cookie. Server-only.
 * Returns null if no session. Role is read from the JWT user_role claim
 * injected by custom_access_token_hook (see supabase/migrations/0011_helper_functions.sql).
 *
 * Cached per request via React.cache — safe to call multiple times in a single RSC render.
 */
export const getSessionUser = cache(async (): Promise<SessionUser | null> => {
  const supabase = await createClient();
  const { data, error } = await supabase.auth.getUser();
  if (error || !data.user) return null;

  // JWT claim is populated by custom_access_token_hook. Fallback to 'user' if missing
  // (e.g., legacy session before hook was enabled).
  const sessionData = await supabase.auth.getSession();
  const jwt = sessionData.data.session?.access_token;
  let role: SessionUser['role'] = 'user';
  if (jwt) {
    try {
      const payload = JSON.parse(
        Buffer.from(jwt.split('.')[1], 'base64').toString('utf-8')
      );
      if (payload.user_role === 'admin' || payload.user_role === 'editor') {
        role = payload.user_role;
      }
    } catch {
      // malformed JWT — treat as 'user'
    }
  }

  return {
    id: data.user.id,
    email: data.user.email ?? null,
    role,
  };
});
```
